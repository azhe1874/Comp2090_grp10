from datetime import datetime
from typing import List

# ========== Item Class ==========
class Item:
    def __init__(self, item_id: str, name: str, unit_price: float, quantity: int = 0):
        self._item_id = item_id
        self._name = name
        self._unit_price = unit_price
        self._quantity = quantity

    def get_item_id(self):
        return self._item_id

    def get_name(self):
        return self._name

    def get_unit_price(self):
        return self._unit_price

    def get_quantity(self):
        return self._quantity

    def set_quantity(self, quantity):
        if quantity >= 0:
            self._quantity = quantity
        else:
            print("Quantity cannot be negative")

    def add_quantity(self, amount):
        self._quantity += amount

    def reduce_quantity(self, amount):
        if amount <= self._quantity:
            self._quantity -= amount
        else:
            print("Insufficient stock")

# ========== Supplier Class ==========
class Supplier:
    def __init__(self, supplier_id: str, name: str, contact: str):
        self._supplier_id = supplier_id
        self._name = name
        self._contact = contact

    def get_supplier_id(self):
        return self._supplier_id

    def get_name(self):
        return self._name

    def get_contact(self):
        return self._contact

# ========== Location Class ==========
class Location:
    def __init__(self, aisle: str, shelf: str, bin: str):
        self._aisle = aisle
        self._shelf = shelf
        self._bin = bin

    def get_aisle(self):
        return self._aisle

    def get_shelf(self):
        return self._shelf

    def get_bin(self):
        return self._bin

# ========== StockRecord Class ==========
class StockRecord:
    def __init__(self, item: Item, warehouse: 'Warehouse', location: Location, quantity: int):
        self._item = item
        self._warehouse = warehouse
        self._location = location
        self._quantity = quantity
        self._last_updated = datetime.now()

    def get_item(self):
        return self._item

    def get_warehouse(self):
        return self._warehouse

    def get_location(self):
        return self._location

    def get_quantity(self):
        return self._quantity

    def set_quantity(self, quantity):
        if quantity >= 0:
            self._quantity = quantity
            self._last_updated = datetime.now()
        else:
            print("Quantity cannot be negative")

# ========== Warehouse Class ==========
class Warehouse:
    def __init__(self, warehouse_id: str, location: str, capacity: int):
        self._warehouse_id = warehouse_id
        self._location = location
        self._capacity = capacity
        self._stock_records: List[StockRecord] = []

    def get_warehouse_id(self):
        return self._warehouse_id

    def get_location(self):
        return self._location

    def get_capacity(self):
        return self._capacity

    def add_stock(self, item: Item, quantity: int, loc: Location):
        # Simple implementation: directly create a new record (ignoring duplicates)
        record = StockRecord(item, self, loc, quantity)
        self._stock_records.append(record)
        print(f"Added {quantity} {item.get_name()}(s) to warehouse")

    def remove_stock(self, item: Item, quantity: int):
        for record in self._stock_records:
            if record.get_item() is item:
                if record.get_quantity() >= quantity:
                    record.set_quantity(record.get_quantity() - quantity)
                    print(f"Removed {quantity} {item.get_name()}(s)")
                    return
                else:
                    print("Insufficient stock")
                    return
        print("Item not found")

# ========== Transaction Class ==========
class Transaction:
    def __init__(self, trans_id: str, item: Item, quantity: int, operator: 'User', trans_type: str):
        self._trans_id = trans_id
        self._item = item
        self._quantity = quantity
        self._operator = operator
        self._type = trans_type  # "in" or "out"
        self._date = datetime.now()

    def get_trans_id(self):
        return self._trans_id

    def get_item(self):
        return self._item

    def get_quantity(self):
        return self._quantity

    def get_operator(self):
        return self._operator

    def get_type(self):
        return self._type

    def get_date(self):
        return self._date

    def process(self):
        if self._type == "in":
            self._item.add_quantity(self._quantity)
            print(f"Inbound {self._quantity} {self._item.get_name()}(s)")
        elif self._type == "out":
            if self._item.get_quantity() >= self._quantity:
                self._item.reduce_quantity(self._quantity)
                print(f"Outbound {self._quantity} {self._item.get_name()}(s)")
            else:
                print("Outbound failed: insufficient stock")

# ========== User Class ==========
class User:
    def __init__(self, user_id: str, name: str, role: str):
        self._user_id = user_id
        self._name = name
        self._role = role  # "admin" or "staff"

    def get_user_id(self):
        return self._user_id

    def get_name(self):
        return self._name

    def get_role(self):
        return self._role

    def can_perform(self, action: str) -> bool:
        # Simple permission: admin can do anything, staff can only inbound/outbound
        if self._role == "admin":
            return True
        elif self._role == "staff":
            return action in ["inbound", "outbound"]
        else:
            return False
